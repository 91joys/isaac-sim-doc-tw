擴充新協定（寫一個 Dissector）#

對應 DeepWiki 7.4 Extending nDPI with New Protocols 與官方 README：How To Add A New Protocol Dissector。

官方的 12 個步驟#

  1. 在 src/include/ndpi_protocol_ids.h 加入新協定與唯一 ID
  2. 在 src/lib/protocols/ 建立新的解析器 .c 檔
  3. 需要跨封包保存的狀態變數，放進 ndpi_typedefs.h：ndpi_flow_tcp_struct（限 TCP）、ndpi_flow_udp_struct（限 UDP）或 ndpi_flow_struct（兩者皆可）
  4. 在 src/include/ndpi_private.h 加入搜尋函式的進入點
  5. 從 src/include/ndpi_define.h 選用一個 selection bitmask（不要改動既有定義）
  6. 在 ndpi_main.c 的 ndpi_init_protocol_defaults 設定協定預設 port
  7. 準備 nBPF 支援：與 nDPI 同層目錄 clone PF_RING，編譯 userland/nbpf
  8. ./autogen.sh && ./configure --with-pcre2（nBPF 與 PCRE2 平常可選，但跑完整測試需要）
  9. make
  10. make check
  11. 更新文件：把新協定加進 doc/protocols.rst
  12. 更新 Visual Studio 設定：把新 .c 檔加進 windows/nDPI.vcxproj

協定註冊#

/* 註冊協定中繼資料 */
int ndpi_set_proto_defaults(
    struct ndpi_detection_module_struct *ndpi_str,
    u_int8_t is_cleartext,       /* 1 = 明文協定 */
    u_int8_t is_app_protocol,    /* 1 = 應用層（HTTP、DNS）、0 = 傳輸層（TLS）*/
    ndpi_protocol_breed_t breed, /* SAFE / ACCEPTABLE / ... */
    u_int16_t protoId,           /* 來自 ndpi_protocol_ids.h */
    char *protoName,
    ndpi_protocol_category_t protoCategory,
    ndpi_protocol_qoe_category_t qoeCategory,
    ndpi_port_range *tcpDefPorts, ndpi_port_range *udpDefPorts,
    u_int8_t is_custom_protocol);

/* 預設 port：最多 5 個，或用範圍 */
ndpi_build_default_ports(ports, 80, 8080, 0, 0, 0);
ndpi_build_default_ports_range(ports, 6881, 6889, 0, 0, 0, 0, 0, 0, 0, 0);

/* 協定層級關係（WhatsApp 可跑在 TLS 或 QUIC 上）*/
ndpi_set_proto_subprotocols(ndpi_str, NDPI_PROTOCOL_WHATSAPP,
    NDPI_PROTOCOL_TLS, NDPI_PROTOCOL_QUIC,
    NDPI_PROTOCOL_NO_MORE_SUBPROTOCOLS);

解析器範本#

static void ndpi_search_myprotocol(struct ndpi_detection_module_struct *ndpi_struct,
                                   struct ndpi_flow_struct *flow) {
  struct ndpi_packet_struct *packet = &ndpi_struct->packet;

  NDPI_LOG_DBG(ndpi_struct, "search myprotocol\n");

  /* 1. 永遠先檢查長度，再碰 payload（記憶體安全鐵則）*/
  if (packet->payload_packet_len < MIN_PROTOCOL_LENGTH) {
    NDPI_EXCLUDE_PROTO(ndpi_struct, flow);  /* 明確不是就早早排除 */
    return;
  }

  /* 2. 比對協定特徵（magic bytes）*/
  if (memcmp(packet->payload, "MYPROTO", 7) == 0) {
    /* 3. 以 DPI 信心設定協定 */
    ndpi_int_add_connection(ndpi_struct, flow,
        NDPI_PROTOCOL_MYPROTOCOL, NDPI_REAL_PROTOCOL);
    return;
  }

  /* 4. 沒比中就排除，之後的封包不會再呼叫這個解析器 */
  NDPI_EXCLUDE_PROTO(ndpi_struct, flow);
}

多階段偵測與延伸解析#

複雜協定（如 TLS）用 flow 狀態欄位分階段：

/* 階段式偵測（TLS 概念示意）*/
if (!flow->tls_quic.client_hello_processed) {
  /* 階段 1：處理 ClientHello，先不設協定 */
  if (process_client_hello(ndpi_struct, flow) == 0) {
    flow->tls_quic.client_hello_processed = 1;
    return;   /* 等更多封包 */
  }
} else if (!flow->tls_quic.certificate_processed) {
  /* 階段 2：ServerHello + 憑證到齊才確認 */
  if (process_server_certificate(ndpi_struct, flow) == 0) {
    flow->tls_quic.certificate_processed = 1;
    ndpi_int_add_connection(ndpi_struct, flow,
        NDPI_PROTOCOL_TLS, NDPI_REAL_PROTOCOL);
  }
}

/* 偵測完成後還要抽 metadata：掛延伸解析 */
static int ndpi_search_myprotocol_extra(struct ndpi_detection_module_struct *s,
                                        struct ndpi_flow_struct *flow) {
  extract_additional_metadata(flow);
  return have_all_metadata(flow) ? 0 : 1;   /* 0 = 停、1 = 續 */
}
/* 在主解析器內： */
flow->extra_packets_func = ndpi_search_myprotocol_extra;

常用輔助工具#

函式／巨集用途
ndpi_int_add_connection()以 DPI 信心設定協定
ndpi_set_detected_protocol()以自訂信心設定協定（如 MATCH_BY_PORT）
ndpi_set_detected_protocol_keeping_master()只更新 app 層、保留 master
NDPI_EXCLUDE_PROTO()排除本解析器（寫入 excluded 位圖）
current_pkt_from_client_to_server()判斷封包方向
ndpi_set_risk()標記安全風險（可附說明訊息）
NDPI_LOG_DBG() / DBG2 / ERR除錯與錯誤記錄
ndpi_lru_find_cache() / ndpi_lru_add_to_cache()快取查詢／寫入

四種偵測手段可以並用#

/* IP 範圍 → 協定 */
ndpi_add_host_ip_subprotocol(ndpi_struct, "192.168.1.0/24", NDPI_PROTOCOL_MYAPP);

/* 主機名稱樣式 → 協定（支援萬用字元與網域層級偵測）*/
ndpi_add_host_url_subprotocol(ndpi_str, "*.example.com",
    NDPI_PROTOCOL_MYAPP, NDPI_PROTOCOL_CATEGORY_WEB, NDPI_PROTOCOL_SAFE, 0);

port 比對信心最低，務必搭配 DPI；字串比對系統會自動處理萬用字元、網域層級與字尾錨定。

測試與驗證#

nDPI 用 PCAP 迴歸測試：測試 PCAP 放 tests/pcap/，預期輸出放 tests/cfgs/default/result/。

# 產生參考輸出（迴歸測試基準）
./example/ndpiReader -i tests/pcap/myprotocol.pcap > \
    tests/cfgs/default/result/myprotocol.pcap.out

# 執行測試套件
cd tests && ./do.sh

驗證清單：

  • 範例流量能正確偵測；相似協定不誤判
  • 畸形封包能優雅處理（fuzzing 是 nDPI CI 的常規項目）
  • 效能可接受（檢查 dissector 呼叫次數與快取命中率）
  • 全部迴歸測試通過（make check）

最佳實務#

  • 記憶體安全：存取 payload 前一律檢查 payload_packet_len；配置一律用 ndpi_malloc / ndpi_free。
  • 早早排除：magic byte 不符就 NDPI_EXCLUDE_PROTO()，避免重複呼叫。
  • 限制封包數：分析 N 個封包後放棄，避免無止境嘗試。
  • 善用快取：昂貴的判斷結果進 LRU 快取。
  • 支援執行期設定：透過 ndpi_struct->cfg 提供開關與參數。
  • 寫文件：註明協定規格（RFC）、偵測啟發式邏輯、已知限制與測試涵蓋。
